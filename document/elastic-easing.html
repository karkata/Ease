function animateEasing(funcEasing) {
    var duration = 1000, old = null, t = 0, p = 0, x = 50, y = 250;
    function loop(time) {
        if (!old) old = time;
        t = (time - old) / duration;
        if (t > 1) t = 1;
        // value = scale * paced + initial
        p = funcEasing(t);
        x = 200 * p + 50;
        y = 250 -200 * p;
        // ball은 움직이는 공을 나타내는 객체입니다.
        ball.center(x, y);
        if (t < 1) {
            requestAnimationFrame(loop);
        }
    }
    requestAnimationFrame(loop);
}
